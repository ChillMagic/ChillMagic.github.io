JitFFI
JIT compiler for Foreign-Function Interface

int function(int a, float b);
ArgTypeList typelist = { &atu_int, &atu_float };

int i = 1, f = 2.0;
ArgDataList dl = { &i, &f };
ArgDataListPartial dlp1 = { &i, nullptr };
ArgDataListPartial dlp2 = { nullptr, &f };

auto f1 = Compile(&function, typelist);
auto f2 = Compile(&function, typelist, dl);
auto f3 = Compile(&function, typelist, dlp1);

int r;

f1(&r, dl);
f2(&r);
f3(&r, dlp2);
